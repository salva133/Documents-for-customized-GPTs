				value = literacy_rate

			}

				value = wealth
				divide = 10
				desc = "QUALIFICATIONS_WEALTH_FACTOR"
			}
			if = {
				limit = {

					desc = "QUALIFICATIONS_FAVORED_TYPE"
				}
			}
			if = {
				limit = {
					is_pop_type = aristocrats
				}
				multiply = {
					value = 5
					desc = "QUALIFICATIONS_FAVORED_TYPE"
				}
			}
			if = {
				limit = {
					culture_accepted = no
				}
				multiply = {
					value = 0.1
					desc = "QUALIFICATIONS_CULTURAL_DISCRIMINATION"

		}
		else_if = {
			limit = {
